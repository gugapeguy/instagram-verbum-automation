**`parseHorario` â€” lĂłgica:**
```javascript
function parseHorario(horarioStr) {
  const now = new Date();
  const today = new Date(now.getFullYear(), now.getMonth(), now.getDate());
  
  // Formato: "amanhĂŁ Xh" ou "amanhĂŁ X:XX"
  if (horarioStr.startsWith('amanhĂŁ')) {
    const timeStr = horarioStr.replace('amanhĂŁ', '').trim();
    const [h, m = 0] = timeStr.replace('h', ':').split(':').map(Number);
    const d = new Date(today);
    d.setDate(d.getDate() + 1);
    d.setHours(h, m, 0, 0);
    return d.getTime();
  }
  
  // Formato: "DD/MM Xh" ou "DD/MM X:XX"
  const dateMatch = horarioStr.match(/(\d{1,2})\/(\d{1,2})\s+(\d{1,2})h?:?(\d{0,2})/);
  if (dateMatch) {
    const [, day, month, h, m = 0] = dateMatch;
    const d = new Date(2026, Number(month) - 1, Number(day), Number(h), Number(m));
    return d.getTime();
  }
  
  // Formato: "HH:MM" ou "HHhMM"
  const timeStr = horarioStr.replace('h', ':');
  const [h, m = 0] = timeStr.split(':').map(Number);
  const d = new Date(today);
  d.setHours(h, m, 0, 0);
  if (d.getTime() <= Date.now()) d.setDate(d.getDate() + 1); // passou: amanhĂŁ
  return d.getTime();
}
```
