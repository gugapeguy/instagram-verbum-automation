**Como** CEO Operações,  
**quero** um script que converta slides HTML para PNG com as dimensões corretas para o Instagram,  
**para que** os criativos gerados pelo CEO Criativo possam ser enviados à Meta Graph API sem necessidade de ferramentas externas.

2. Função `exportSlides(htmlPath, options)` exportada: aceita path do HTML, número de slides e formato (`carrossel` | `feed` | `stories` | `reels`)

## Dev Notes

**Playwright — estrutura básica:**
```javascript
const { chromium } = require('playwright');
const path = require('path');
const fs = require('fs');

async function exportSlides(htmlPath, options = {}) {
  const { numSlides = 1, format = 'carrossel', outputDir, slug } = options;
  
  const browser = await chromium.launch();
  const page = await browser.newPage();
  
  const viewport = ['stories', 'reels'].includes(format)
    ? { width: 1080, height: 1920 }
    : { width: 1080, height: 1080 };
  
  await page.setViewportSize(viewport);
  await page.goto('file:///' + path.resolve(htmlPath), { waitUntil: 'networkidle' });
  
  const dir = outputDir || path.join('criatives', 'rendered', slug || 'output');
  fs.mkdirSync(dir, { recursive: true });
  
  if (format === 'carrossel') {
    const slides = await page.$$('.slide');
    for (let i = 0; i < slides.length; i++) {
      const filename = `slide-${String(i + 1).padStart(2, '0')}.png`;
      await slides[i].screenshot({ path: path.join(dir, filename) });
    }
    console.log(`✅ Export concluído: ${slides.length} slides → ${dir}/`);
  } else {
    await page.screenshot({ path: path.join(dir, 'slide-01.png') });
    console.log(`✅ Export concluído: 1 slide → ${dir}/`);
  }
  
  await browser.close();
}

module.exports = { exportSlides };
```

**Instalação do Playwright:**
```bash
npm install --save-dev playwright
npx playwright install chromium
```
